function setup() {
  let w = 700
  createCanvas(w, w)
  background(0)

  let r = random

  push()
  translate(w / 2, w / 2)

  let s = r(50, 250)
  let p = r(8, 13)

  for (let i = p; i < w * 2; i += p) {
    stroke(i % s, 240)
    fill(0, 2)
    ellipse(0, 0, i, i * 2)
    rotate(0.1)
  }
  pop()

  fill(r(30, 250), r(30, 250), r(30, 250), r(50, 80))
  rect(0, 0, w, w)
}
